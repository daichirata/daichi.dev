RustгҒ гҒ‘гҒ§Ruby native extensionгӮ’жӣёгҒҸ

гҒ“гҒ®иЁҳдәӢгҒҜ Rust Advent Calendar 2015 10ж—Ҙзӣ®гҒ®иЁҳдәӢгҒ§гҒҷгҖӮ

RustгҒҜе…ұжңүгғ©гӮӨгғ–гғ©гғӘгӮ’з”ҹжҲҗгҒ—гҒҰCгҒ®гғ—гғӯгӮ°гғ©гғ гҒЁгғӘгғігӮҜгҒҷгӮӢгҒ“гҒЁгҒҢеҸҜиғҪгӮүгҒ—гҒ„гҒ®гҒ§гҖҒRubyгҒӢгӮүrequireгҒ—гҒҰгҒЎгӮғгӮ“гҒЁеӢ•гҒҸnative extensionгӮ’RustгҒ гҒ‘гӮ’дҪҝгҒЈгҒҰжӣёгҒ„гҒҰгҒҝгӮӢгҖӮ

ж—ўгҒ«дјјгҒҹгӮҲгҒҶгҒӘгҒ“гҒЁгӮ’гӮ„гҒЈгҒҰгҒ„гӮӢдҫӢгҒҜгҒӮгӮӢгӮ“гҒ гҒ‘гҒ©гҖҒCгғ•гӮЎгӮӨгғ«гӮ’з”Ёж„ҸгҒ—гҒҰгҒқгҒ“гҒӢгӮүRustгӮ’е‘јгҒіеҮәгҒҷеҪўгҒ«гҒӘгҒЈгҒҰгҒ„гӮӢгҒ®гҒ§гҖҒд»ҠеӣһгҒҜCгӮ’жӣёгҒӢгҒҡгҒ«гӮ„гҒЈгҒҰгҒҝгҒҹгҒ„гҒЁжҖқгҒҶгҖӮгҒЎгҒӘгҒҝгҒ«е®ҹз”ЁжҖ§гҒҜзү№гҒ«з„ЎгҒ„гҒ®гҒ§гҒӮгҒ—гҒӢгӮүгҒҡгҖӮ

Dylib

RustгҒ§е…ұжңүгғ©гӮӨгғ–гғ©гғӘгӮ’дҪңжҲҗгҒ—гҒҰCгҒӢгӮүе‘јгҒіеҮәгҒҷгҒ«гҒҜгҖҒд»ҘдёӢгҒ®гӮҲгҒҶгҒ«гҒҷгӮҢгҒ°гӮҲгҒ„гҖӮ

FFI

RustгҒӢгӮүCгӮ’е‘јгҒіеҮәгҒҷгҒ«гҒҜRustгҒ®д»–иЁҖиӘһй–ўж•°гӮӨгғігӮҝгғјгғ•гӮ§гғјгӮ№пјҲFFIпјүгҒ®гӮ¬гӮӨгғү - QiitaгҒ«ж—Ҙжң¬иӘһиЁігҒҢгҒӮгӮӢгҒ®гҒ§гҒқгӮҢгӮ’гҒҝгӮӢгҒ®гҒҢгӮҸгҒӢгӮҠгӮ„гҒҷгҒ„гҒЁжҖқгҒҶгҖӮ

#[link(name = "snappy")]
extern {
    fn snappy_max_compressed_length(source_length: size_t) -> size_t;

}

VALUEгҒҜRubyгҒ®гғҮгғјгӮҝгӮ’жүұгҒҶCеҒҙгҒ®еһӢгҒ§гҖҒж§ӢйҖ дҪ“гҒ«еҜҫгҒҷгӮӢгғқгӮӨгғігӮҝеҸҲгҒҜеҖӨгҒқгҒ®зү©гҖӮrb_cObjectгҒҜCеҒҙгҒ®еӨ–йғЁеӨүж•°гҒӘгҒ®гҒ§гҖҒRustгҒӢгӮүеҸӮз…§гҒҷгӮӢгҒ«гҒҜexternгғ–гғӯгғғгӮҜеҶ…гҒ§staticгҒЁгҒ—гҒҰе®ҡзҫ©гҒҷгӮӢгҖӮд»ҠеӣһгҒҜжӣёгҒҚжҸӣгҒҲгӮӢеҝ…иҰҒгҒҜз„ЎгҒ„гҒ®гҒ§*mutгҒҜгҒӨгҒ‘гҒӘгҒ„гҖӮrb_define_classгҒҜCеҒҙгҒ§гҒҜ

#![allow(non_snake_case, non_camel_case_types)]

extern crate libc;
mod fib;

use std::ffi::CString;

type VALUE = libc::c_ulong;

extern {
    static rb_cObject: VALUE;

    fn rb_define_class(name: *const libc::c_char, rb_super: VALUE) -> VALUE;

    fn rb_define_method(klass: VALUE,
                        name: *const libc::c_char,
                        func: extern fn(v: VALUE, v2: VALUE),
                        argc: libc::c_int) -> libc::c_void;

    fn rb_num2long(val: VALUE) -> libc::c_long;
}

fn rb_int2fix(num: u32) -> VALUE {
    return ((num as VALUE) << 1) | 0x01;
}

extern fn rb_fib(_: VALUE, rb_num: VALUE) {
    let num = unsafe { rb_num2long(rb_num) } as u32;
    let result = fib::fib(num);

    rb_int2fix(result);
}

#[no_mangle]
pub extern fn Init_rust() {
    let c_name = CString::new("Rust").unwrap();
    let fib = CString::new("fib").unwrap();

    unsafe {
        let rb_cRust = rb_define_class(c_name.as_ptr(), rb_cObject);

        rb_define_method(rb_cRust, fib.as_ptr(), rb_fib, 1);
    }
}

CгҒ®гғҳгғғгғҖгғјгҒӢгӮүRustгҒ®е®ҡзҫ©гғ•гӮЎгӮӨгғ«гӮ’з”ҹжҲҗгҒҷгӮӢcrabtw/rust-bindgenгҒЁгҒ„гҒҶгғ„гғјгғ«гҒҢдёҖеҝңгҒӮгҒЈгҒҰгҖҒгҒ“гҒ®гғ„гғјгғ«гӮ’е…ғгҒ«RubyгҒ®гғҳгғғгғҖгғјгғ•гӮЎгӮӨгғ«гҒ®е®ҡзҫ©гӮ’еҮәеҠӣгҒҷгӮӢгҒЁгҒ“гҒҶгҒӘгӮӢгҖӮе®Ңз’§гҒӘеӨүжҸӣгҒҫгҒ§гҒҜйӣЈгҒ—гҒҸгҒҰгҖҒгҒқгҒ®гҒҫгҒҫгҒ§гҒҜгӮЁгғ©гғјгҒ«гҒӘгҒЈгҒҰдҪҝгҒҶгҒ“гҒЁгҒҢеҮәжқҘгҒӘгҒ„гӮ“гҒ гҒ‘гҒ©гҖҒзөҗж§ӢеҸӮиҖғгҒ«гҒӘгӮӢгҒЁжҖқгҒҶгҒ®гҒ§дёҖж—ҰеҮәеҠӣгҒ—гҒҰгҒ“гҒ®гғ•гӮЎгӮӨгғ«гӮ’иҰӢгҒӘгҒҢгӮүгӮ„гӮӢгҒЁжҚ—гӮӢгҒЁжҖқгҒҶгҖӮ

гҒҹгҒ гҖҒеҪ“з„¶гҒӘгӮ“гҒ гҒ‘гҒ©CгҒ®гғһгӮҜгғӯгҒ«гҒҜеҜҫеҝңгҒ—гҒҰгҒ„гҒӘгҒ„гҒ®гҒ§гҖҒеҜҫеҝңгҒҷгӮӢCгҒ®й–ўж•°гҒҢгҒӮгӮҢгҒ°иүҜгҒ„гӮ“гҒ гҒ‘гҒ©з„ЎгҒӢгҒЈгҒҹе ҙеҗҲгҒ«гҒҜиҮӘеҲҶгҒ§е®ҹиЈ…гҒҷгӮӢеҝ…иҰҒгҒҢгҒӮгӮӢгҖӮдҫӢгҒҲгҒ°CгҒ®intгӮ’VALUEгҒ«еӨүжҸӣгҒҷгӮӢFIX2LONGгғһгӮҜгғӯгҒЁгҒӢгҖӮRubyгҒҜжҜ”ијғзҡ„еҜҫеҝңгҒҷгӮӢй–ўж•°гҒҢеӨҡгҒ„ж°—гҒҢгҒҷгӮӢгҒ®гҒ§ж„ҸеӨ–гҒЁгҒӘгӮ“гҒЁгҒӢгҒӘгӮӢгҒӢгӮӮпјҹ

еҫҢгҖҒRustгҒ®еҸҜеӨүй•·еј•ж•°гҒёгҒ®еҜҫеҝңгӮӮгҒҷгҒ”гҒҸеҫ®еҰҷгҒ§гӮ„гӮҠж–№гҒҢгӮҸгҒӢгӮүгҒӘгҒӢгҒЈгҒҹгҒ®гҒ§гҖҒдёҠгҒ®дҫӢгҒ§гҒҜгҒ”гҒҫгҒӢгҒ—гҒҰгҒ„гӮӢйғЁеҲҶгҒҢгҒӮгӮӢгҖӮrb_define_methodгҒЁгҒ„гҒҶCеҒҙгҒ®й–ўж•°гҒҜ3з•Әзӣ®гҒ®еј•ж•°гҒ«й–ўж•°гғқгӮӨгғігӮҝгӮ’еҸ—гҒ‘еҸ–гӮӢгӮ“гҒ гҒ‘гҒ©гҖҒеј•ж•°гҒҢеҸҜеӨүй•·еј•ж•°гҒЁгҒ—гҒҰе®ҡзҫ©гҒ•гӮҢгҒҰгҒ„гӮӢгҖӮRustгҒ§гӮӮexternгғ–гғӯгғғгӮҜеҶ…гҒ®й–ўж•°гӮ·гӮ°гғҚгғҒгғЈгҒ«гҒҜеҸҜеӨүй•·гҒ«е®ҡзҫ©гҒ§гҒҚгӮӢгҒЈгҒҪгҒ„гҒ®гҒ§еҲқгӮҒгҒҜ

extern {
    fn rb_define_method(klass: VALUE,
                        name: *const libc::c_char,
                        func: extern fn(v: VALUE, ...),
                        argc: libc::c_int) -> libc::c_void;
}

гҒЁе®ҡзҫ©гҒ—гҒҰгҒ„гҒҹгӮ“гҒ гҒ‘гҒ©гҖҒrb_fibгӮ’жёЎгҒ—гҒҰгҒ„гӮӢйғЁеҲҶгҒ§еһӢгҒҢдёҖиҮҙгҒ—гҒӘгҒ„гҒЁгҒӢгҒ§гӮігғігғ‘гӮӨгғ«гҒҢйҖҡгӮүгҒӘгҒ„гҒ®гҒ§зөҗеұҖгҒӮгҒҚгӮүгӮҒгҒҰгҖҒ2гҒӨгҒ®еј•ж•°гӮ’еҸ—гҒ‘еҸ–гӮӢй–ўж•°гҒ®гғқгӮӨгғігӮҝгҒЁгҒ—гҒҰе®ҡзҫ©гҒ—гҒҰгҒҠиҢ¶гӮ’гҒ«гҒ”гҒ—гҒҰгҒ„гӮӢгҖӮbindgenгҒ§з”ҹжҲҗгҒ•гӮҢгӮӢеҘҙгҒ§гӮӮгӮҲгҒҸгӮҸгҒӢгӮүгҒӘгҒӢгҒЈгҒҹгҒ®гҒ§гҒ“гҒ®иҫәгҒ«и©ігҒ—гҒ„дәәжҳҜйқһж•ҷгҒҲгҒҰдёӢгҒ•гҒ„гҖӮ

link attributeгӮ’дҪҝгҒЈгҒҰжӯЈж”»жі•гҒ§гғ“гғ«гғүгҒҷгӮӢгҒ«гҒҜгҖҒrubyгӮ’е…ұжңүгғ©гӮӨгғ–гғ©гғӘд»ҳгҒҚгҒ§гғ“гғ«гғүгҒ—гҒҹдёҠгҒ§build.rsгӮ’иҝҪеҠ гҒҷгӮҢгҒ°иүҜгҒ„гҖӮ

$ RUBY_CONFIGURE_OPTS="--enable-shared" rbenv install 2.2.3
  • build.rs
use std::process::Command;

fn main() {
    let output = Command::new("ruby")
        .arg("-e")
        .arg("puts RbConfig::CONFIG['libdir']")
        .output()
        .unwrap_or_else(|e| { panic!("failed to execute process: {}", e) });

    println!("cargo:rustc-link-search=native={}", String::from_utf8_lossy(&output.stdout));
}

еҫҢгҒҜгҖҒexternгҒ®еүҚгҒ®иЎҢгҒ«#[link(name = "ruby")]гҒЁжӣёгҒ‘гҒ°cargo buildгҒ§гғ“гғ«гғүгҒ§гҒҚгӮӢгҖӮгҒҹгҒ гҖҒlibrubyгҒ«гғӘгғігӮҜгҒ•гӮҢгҒҰгҒ„гҒҰдҫқеӯҳй–ўдҝӮгҒҢгҒӮгӮӢжүҖгҒ гҒ‘жіЁж„ҸгҒ—гҒҰгҒҠгҒ„гҒҹгҒ»гҒҶгҒҢгҒ„гҒ„гҖӮ
